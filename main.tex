%----------------------------------------------------------------------------------------
%   PACKAGES AND OTHER DOCUMENT CONFIGURATIONS
%----------------------------------------------------------------------------------------

\documentclass[final]{beamer}

% Use the beamerposter package for laying out the poster
% Edit this scale to change the font size, and all rendered graphics
\usepackage[scale=1.55]{beamerposter}
\usepackage{import}

% Allow for larger equations
\usepackage{lmodern}
\usepackage{exscale} 

% Use the confposter theme supplied with this template
% Look in a sub folder: http://tex.stackexchange.com/a/284157
\usepackage{packages/beamerthemeconfposter}

% Define udel colors (from udel main website)
\definecolor{udelblue}{RGB}{0, 72, 165}
\definecolor{udelgold}{RGB}{255, 211, 0}

% Many more colors are available for use in beamerthemeconfposter.sty
\setbeamercolor{headline title}{fg=white,bg=udelblue} % Colors of the block titles
\setbeamercolor{headline line}{fg=black,bg=udelgold} % Colors of the block titles
\setbeamercolor{block title}{fg=udelblue,bg=white} % Colors of the block titles
\setbeamercolor{block body}{fg=black,bg=white} % Colors of the body of blocks
\setbeamercolor{block alerted title}{fg=black,bg=udelgold!85} % Colors of the highlighted block titles
\setbeamercolor{block alerted body}{fg=black,bg=udelgold!10} % Colors of the body of highlighted blocks

% set colors for itemize/enumerate
\setbeamercolor{item}{fg=udelgold}
\setbeamercolor{item projected}{fg=black,bg=udelgold}

%-----------------------------------------------------------
% Define the column widths and overall poster size
% To set effective sepwid, onecolwid and twocolwid values, first choose how many columns you want and how much separation you want between columns
% In this template, the separation width chosen is 0.024 of the paper width and a 4-column layout
% onecolwid should therefore be (1-(# of columns+1)*sepwid)/# of columns e.g. (1-(4+1)*0.024)/4 = 0.22
% Set twocolwid to be (2*onecolwid)+sepwid = 0.464
% Set threecolwid to be (3*onecolwid)+2*sepwid = 0.708

\newlength{\sepwid}
\newlength{\onecolwid}
\newlength{\twocolwid}
\newlength{\threecolwid}
\setlength{\paperwidth}{47in} % A0 width: 46.8in
\setlength{\paperheight}{33in} % A0 height: 33.1in
\setlength{\sepwid}{0.024\paperwidth} % Separation width (white space) between columns
\setlength{\onecolwid}{0.22\paperwidth} % Width of one column
\setlength{\twocolwid}{0.464\paperwidth} % Width of two columns
\setlength{\threecolwid}{0.708\paperwidth} % Width of three columns
\setlength{\topmargin}{-0.5in} % Reduce the top margin size
%-----------------------------------------------------------

\usepackage{graphicx}  % Required for including images
\usepackage{tikz} % Allow placing of logo in top right
\usepackage{booktabs} % Top and bottom rules for tables

%----------------------------------------------------------------------------------------
%   TITLE SECTION 
%----------------------------------------------------------------------------------------

\title{Unnecessarily Complicated Research Title} % Poster title

\author{John Smith, James Smith and Jane Smith} % Author(s)

\institute{University of Delaware - Department of Mechanical Engineering} % Institution(s)

%----------------------------------------------------------------------------------------

\begin{document}

\addtobeamertemplate{block end}{}{\vspace*{2ex}} % White space under blocks
\addtobeamertemplate{block alerted end}{}{\vspace*{2ex}} % White space under highlighted (alert) blocks

\setlength{\belowcaptionskip}{2ex} % White space under figures
\setlength\belowdisplayshortskip{2ex} % White space under equations

% The whole poster is enclosed in one beamer frame
\begin{frame}[t]

%----------------------------------------------------------------------------------------


% Add logo in top right
\tikz[remember picture,overlay]
  \node[anchor=north east,inner sep=0in,shift={(-2cm,-3.2cm)}] at (current page.north east)
    {\includegraphics[height=2in]{figures/udel_logo.png}};

%----------------------------------------------------------------------------------------


% Add some padding above the first row
\vspace{0mm}

% Begin the first row of blocks
\begin{columns}[t,totalwidth=\paperwidth]

    % Empty spacer column
    \begin{column}{\sepwid}\end{column}

    % The first block (block 1.1)
    \begin{column}{\onecolwid}\vspace{-.6in}
        \input{sections/block_1_1}
    \end{column} % End of column 2.1
    
    % The second block (block 1.2)
    \begin{column}{\onecolwid}\vspace{-.6in}
        \input{sections/block_1_2}
    \end{column}
    
    % The third block (block 1.3)
    \begin{column}{\onecolwid}\vspace{-.6in}
        \input{sections/block_1_3}
    \end{column}
    
    % Empty spacer column
    \begin{column}{\sepwid}\end{column}

\end{columns}

% Force the content to be at the bottom of the page
\vfill

% Begin the second row of blocks
\begin{columns}[t,totalwidth=\paperwidth]

    % Empty spacer column
    \begin{column}{\sepwid}\end{column}

    % The first block (block 2.1)
    \begin{column}{\onecolwid}\vspace{-.6in}
        \input{sections/block_2_1}
    \end{column} % End of column 2.1
    
    % The second block (block 2.2)
    \begin{column}{\onecolwid}\vspace{-.6in}
        \input{sections/block_2_2}
    \end{column}
    
    
    % The third block (block 2.3)
    \begin{column}{\onecolwid}\vspace{-.6in}
        \input{sections/block_2_3}
    \end{column}
    
    % The forth block (block 2.4)
    \begin{column}{\onecolwid}\vspace{-.6in}
        \input{sections/block_2_4}
    \end{column}
    
    % Empty spacer column
    \begin{column}{\sepwid}\end{column}

\end{columns}


%----------------------------------------------------------------------------------------


\end{frame} % End of the enclosing frame


\end{document}
